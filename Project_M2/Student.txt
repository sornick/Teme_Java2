package students;

public class Student {
                private String  name; 
	private String  cnp;
	private Integer id;

	public Student(Integer id, String name, String cnp)
	{
	    this.id = id;
		this.name = name;
		this.cnp  = cnp;
	}
	
	public String getName() 
	{
		return name;
	}

	public String getCNP()
	{
		return cnp;
	}
	
	public Integer getID() 
	{
		return id;
	}
	
}
